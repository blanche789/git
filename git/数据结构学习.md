###  数据结构学习
- 二叉树遍历递归算法
	
		//存储结构
		typedef struct BiTNode{
			TElemType data;
			struct BiTNode *lchild,*rchild;
		
		}BiTNode,*BiTree;
		//先序
		void PreOrder(BiTree T){
			printf("%d",T->data)；
			PreOrder(T->lchild);
			PreOrder(T->rchild);
		}
		//中序
		void InOrder(BiTree T){
			InOrder(T->lchild);
			printf("%d",T->data)；
			InOrder(T->rchild);
		
		}
		//后序
		void PosOrder(BiTree T){
			PosOrder(T->lchild);
			PosOrder(T->rchild);
			printf("%d",T->data)；
		
		}
